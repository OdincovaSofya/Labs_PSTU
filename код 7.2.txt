include "pch.h"
#include <iostream>
#include <string>

using namespace std;

int sum(int a, ...)
{
	int *b = &a; 
	int s = 0;
	for (; a != 0; a--)
		s += *(++b);
	return s;
}

int main()
{
	setlocale(LC_ALL, "Rus");
	cout <<"Сумма чисел (5, 1, 2) = " << sum(3, 5, 1, 2) << endl;
	cout << "\nСумма чисел (1, 2, 3, 4, 5 ,6, 7) = " << sum(7,1, 2, 3,4,5,6,7) << endl;
	cout << "\nСумма чисел (8, 10, 2, 4 ,5, 1, 9, 2, 4, 0, 7) = " << sum(11, 8, 10, 2, 4,5,1,9,2,4,0,7) << endl;
}
